cmake_minimum_required(VERSION 3.0.0)
project(ArtificialNN VERSION 0.1.0)

set(ANN_INCLUDE_DIR ${ANN_INCLUDE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/ArtificialNN/src/CNN")
set(ANN_INCLUDE_DIR ${ANN_INCLUDE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/ArtificialNN/src/CNN/LearnCnn")
set(ANN_INCLUDE_DIR ${ANN_INCLUDE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/ArtificialNN/src/Instruments/Filter")
set(ANN_INCLUDE_DIR ${ANN_INCLUDE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/ArtificialNN/src/Instruments/Functors")
set(ANN_INCLUDE_DIR ${ANN_INCLUDE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/ArtificialNN/src/Instruments/Matrix")
set(ANN_INCLUDE_DIR ${ANN_INCLUDE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/ArtificialNN/src/Instruments/Weights")
set(ANN_INCLUDE_DIR ${ANN_INCLUDE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/ArtificialNN/src/Perceptron")
set(ANN_INCLUDE_DIR ${ANN_INCLUDE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/ArtificialNN/src/Perceptron/learnPerc")
set(ANN_INCLUDE_DIR ${ANN_INCLUDE_DIR} "${CMAKE_CURRENT_SOURCE_DIR}/ArtificialNN/src/Neyron")

if(${CMAKE_CXX_COMPILER_ID} STREQUAL MSVC)
	message(-- Building <ArtificialNN> for <${CMAKE_CXX_COMPILER_ID}> compiler)
	SET(CMAKE_CXX_FLAGS  "${CMAKE_CXX_FLAGS} /arch:SSE /arch:SSE2 /arch:AVX")
	SET(BUILD_SHARED_LIBS OFF)

elseif (${CMAKE_CXX_COMPILER_ID} STREQUAL AppleClang)
	message(-- Building <ArtificialNN> for <${CMAKE_CXX_COMPILER_ID}> compiler)
	SET(CMAKE_CXX_FLAGS  "${CMAKE_CXX_FLAGS} -std=c++17")
elseif (${CMAKE_CXX_COMPILER_ID} STREQUAL GNU)
	message(-- Building <ArtificialNN> for <${CMAKE_CXX_COMPILER_ID}> compiler)
	SET(CMAKE_CXX_FLAGS  "${CMAKE_CXX_FLAGS} -std=c++17 --param ggc-min-expand=1 --param ggc-min-heapsize=32768")
else()
	message(-- Building <ArtificialNN> for ${CMAKE_CXX_COMPILER_ID} compiler)
	SET(CMAKE_CXX_FLAGS  "${CMAKE_CXX_FLAGS} -std=c++17")
endif()
